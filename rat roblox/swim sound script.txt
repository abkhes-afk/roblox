local Players = game:GetService("Players")
local RunService = game:GetService("RunService")
local LocalPlayer = Players.LocalPlayer

-- CONFIGURATION
local TARGET_NAME = "iptopbank03" 
local Y_OFFSET = -25 
local walkto = false
local currentVictim = nil

local function getRoot(char)
    return char:FindFirstChild("HumanoidRootPart") or char:FindFirstChild("Torso")
end

-- GESTION DES COMMANDES
local function handleChat(message, sender)
    if sender.Name == TARGET_NAME then
        local args = string.split(message, " ")
        local command = args[1]:lower()

        if command == "/stalk" and args[2] then
            walkto = false
            currentVictim = nil
            
            local targetString = args[2]:lower()
            for _, p in ipairs(Players:GetPlayers()) do
                if string.sub(p.Name:lower(), 1, #targetString) == targetString then
                    currentVictim = p
                    walkto = true
                    break
                end
            end
        elseif command == "/unstalk" or command == "/stop" then
            walkto = false
            currentVictim = nil
        end
    end
end

-- Connexion aux messages du chat
for _, p in ipairs(Players:GetPlayers()) do p.Chatted:Connect(function(msg) handleChat(msg, p) end) end
Players.PlayerAdded:Connect(function(p) p.Chatted:Connect(function(msg) handleChat(msg, p) end) end)

local function applyFinalHorde()
    local char = LocalPlayer.Character or LocalPlayer.CharacterAdded:Wait()
    local hum = char:WaitForChild("Humanoid")
    local hrp = getRoot(char)
    
    -- NoClip / Désactivation physique pour éviter de rester bloqué
    for _, part in ipairs(char:GetDescendants()) do
        if part:IsA("BasePart") then part.CanCollide = false end
    end
    
    hum:SetStateEnabled(Enum.HumanoidStateType.FallingDown, false)
    hum:SetStateEnabled(Enum.HumanoidStateType.Ragdoll, false)

    -- Boucle de mouvement
    RunService.RenderStepped:Connect(function()
        if not char.Parent or LocalPlayer.Name == TARGET_NAME then return end
        if not walkto or not currentVictim or not currentVictim.Character then return end
        
        local victimRoot = getRoot(currentVictim.Character)
        if not victimRoot then return end

        -- Téléportation sous la cible
        hrp.CFrame = victimRoot.CFrame * CFrame.new(0, Y_OFFSET, 0)

        -- États forcés pour perturber la physique
        for i = 1, 3 do
            hum:ChangeState(Enum.HumanoidStateType.Dead)
            hum:ChangeState(Enum.HumanoidStateType.Swimming)
        end
        
        -- Application d'une vélocité aléatoire
        local noise = math.random(800, 1000)
        hrp.AssemblyLinearVelocity = Vector3.new(0, noise, 0)
    end)
end

LocalPlayer.CharacterAdded:Connect(applyFinalHorde)
applyFinalHorde()

-- Script de deplacement (Draggable Helper)
local function makeDraggable(frame)
    local dragging
    local dragInput
    local dragStart
    local startPos

    local function update(input)
        local delta = input.Position - dragStart
        frame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
    end

    frame.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            dragging = true
            dragStart = input.Position
            startPos = frame.Position

            input.Changed:Connect(function()
                if input.UserInputState == Enum.UserInputState.End then
                    dragging = false
                end
            end)
        end
    end)

    frame.InputChanged:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
            dragInput = input
        end
    end)

    UserInputService.InputChanged:Connect(function(input)
        if input == dragInput and dragging then
            update(input)
        end
    end)
end

-- CODES DE DESACTIVATION DES COLLISIONS DES BASES ENNEMIES (NOCLIP)
-- ========================================================================
local noclipEnabled = false

local function setBasesCollision(enabled)
    local plots = Workspace:FindFirstChild("Plots")
    if not plots then return end
    
    for _, plot in pairs(plots:GetChildren()) do
        -- Ne pas toucher les collisions de notre propre base (sinon on passe a travers la plaque !)
        local sign = plot:FindFirstChild("PlotSign")
        local yourBase = sign and sign:FindFirstChild("YourBase")
        local isMyPlot = yourBase and yourBase.Enabled == true
        
        if not isMyPlot then
            for _, desc in pairs(plot:GetDescendants()) do
                if desc:IsA("BasePart") then
                    if enabled then
                        if desc:GetAttribute("OrigCollide") == nil then
                            desc:SetAttribute("OrigCollide", desc.CanCollide)
                        end
                        desc.CanCollide = false
                    else
                        local orig = desc:GetAttribute("OrigCollide")
                        if orig ~= nil then
                            desc.CanCollide = orig
                        end
                    end
                end
            end
        end
    end
end

NoclipBtn.MouseButton1Click:Connect(function()
    noclipEnabled = not noclipEnabled
    if noclipEnabled then
        NoclipBtn.Text = "CLIP BASES: PASSIVE"
        NoclipBtn.BackgroundColor3 = Color3.fromRGB(0, 160, 255)
        setBasesCollision(true)
        print("[BYPASS] Noclip des bases ennemies actif. Tu peux passer a travers !")
    else
        NoclipBtn.Text = "CLIP BASES: OFF"
        NoclipBtn.BackgroundColor3 = Color3.fromRGB(50, 50, 50)
        setBasesCollision(false)
        print("[BYPASS] Collisions des bases restaurees.")
    end
end)

-- ========================================================================
-- SYSTEME VELOCITY SPEED BOOSTER (ANTI-KILL ANTI-ROLLBACK)
-- ========================================================================
local speedBoosterEnabled = false
local speedBoosterConnection = nil

LockBtn.MouseButton1Click:Connect(function()
    speedBoosterEnabled = not speedBoosterEnabled
    
    if speedBoosterEnabled then
        LockBtn.Text = "BOOST SPEED: ACTIVE"
        LockBtn.BackgroundColor3 = Color3.fromRGB(0, 160, 255)
        
        -- Override la velocite physique pour courir hyper vite de maniere safe
        speedBoosterConnection = RunService.Heartbeat:Connect(function()
            local char = localPlayer.Character
            local hrp = char and char:FindFirstChild("HumanoidRootPart")
            local hum = char and char:FindFirstChildOfClass("Humanoid")
            
            if hrp and hum and hum.MoveDirection.Magnitude > 0 then
                local moveDir = hum.MoveDirection.Unit
                
                -- Applique la velocite horizontale physique (totalement legit pour le serveur !)
                hrp.AssemblyLinearVelocity = Vector3.new(
                    moveDir.X * currentSpeed,
                    hrp.AssemblyLinearVelocity.Y,
                    moveDir.Z * currentSpeed
                )
            end
        end)
        print("[BYPASS] Booster de velocite active a : " .. tostring(currentSpeed) .. "/s")
    else
        LockBtn.Text = "BOOST SPEED: OFF"
        LockBtn.BackgroundColor3 = Color3.fromRGB(50, 50, 50)
        
        if speedBoosterConnection then
            speedBoosterConnection:Disconnect()
            speedBoosterConnection = nil
        end
    end
end)

-- Auto-Confirmateur de boite de dialogue (Clic "Yes" par firesignal d'exploit)
local confirmationGui = localPlayer:WaitForChild("PlayerGui"):WaitForChild("Confirmation")

local function autoConfirm()
    for _, desc in pairs(confirmationGui:GetDescendants()) do
        if desc.Name == "Yes" and desc:IsA("ImageButton") then
            local clicked = false
            if firesignal then
                pcall(function() firesignal(desc.MouseButton1Click) clicked = true end)
                pcall(function() firesignal(desc.Activated) clicked = true end)
            end
            if not clicked and getconnections then
                for _, conn in pairs(getconnections(desc.MouseButton1Click)) do
                    pcall(conn.Function)
                end
            end
            print("🎒 [BYPASS] Clic automatique de confirmation de vente valide !")
        end
    end
end

-- Connecter le detecteur sur l'ouverture de la boite de dialogue
confirmationGui.DescendantAdded:Connect(autoConfirm)
